useradd = Add new User

example = sudo useradd ali

usermod = Modify an existing user

example =sudo usermod -aG sudo ali

passwd = set or change user's password

example = sudo passwd ali

who = show all currently logged-in users

example = who

whoami = Show the currently logged-in user

example= whoami

id = Show user ID (UID), group ID (GID), and group memberships 

example = id
