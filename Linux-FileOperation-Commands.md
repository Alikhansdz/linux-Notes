cp= copy Files 

example= cp file.txt /home/kali/Desktop

cp -r = copy diectories Reursively

mv = move or Rename the file

example=  mv file.txt /home/kali/Downloads

rm = Delete the files

example = rm file.txt

rm -r = Delete Directories and their Contents

rmdir = Delete Empty Directories 

example= rmdir movies

mkdir = make a new Directory

example = mkdir movies

mkdir -p = Create Nested Directories

example = mkdir -p /Desktop/movies/3-idiots

touch = Create an Empty File OR change its timestamp

example= touch file.txt

stat = Display Detailed File or Directory Info

example =stat Desktop
